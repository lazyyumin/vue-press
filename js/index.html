# 防抖函数

  • 防抖是在触发事件以后，一定时间内只执行最后一次函数的操作
function debounce(callback, wait = 1000) {
  let timer = null
  
  return function() {
    let args = arguments
    if(timer) clearTimeout(timer)
    timer = setTimeout(()=> {
      callback.apply(this, args)
    }, wait)
  }
}

# 节流函数

function throttle(callback, wait = 1000) {
  let pre = 0
  return function() {
    let now = Date.now()
    let arg = arguments
    if(now - pre > wait) {
      callback.apply(this, arg)
      pre = now
    }
  }
}

# 深拷贝

function deepClone(obj, target = new Map()) {
  //  基本数据类型处理
  if(obj === null || typeof obj != 'object') {
    return obj
  }
  //  Date和正则处理
  if(obj instanceof Date) return new Date(obj)
  if(obj instanceof RegExp) return new RegExp(obj)
  //
  if(target.has(obj)) {
    return target.get(obj)
  }
  //  缓存对象
  target.set(obj)

  let result = Array.isArray(obj) ? [] : {}
  for(let key in obj) {
    res[key] = deepClone(obj[key], target)
  }
  return result
}

或者是使用JSON

let result = JSON.parse(JSON.stringify(obj))

缺点是不能处理undefined和函数等

# 冒泡排序

function bubbleSort(arr) {
  for(let i=0; i<arr.length-1; i++) {
    for(let j=0; j< arr.length-1-i; j++) {
      if(arr[i]> arr[j]) {
        let temp = arr[i]
        arr[i] = arr[j]
        arr[j] = temp
      }
    }
  }
  return arr
}
